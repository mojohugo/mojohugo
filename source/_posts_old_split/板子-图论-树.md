倍增lca,树上距离:
vector<int>g[N];
int f[N][20], dep[N];
void dfs(int u, int fa) {
    f[u][0] = fa;
    dep[u] = dep[fa] + 1;
    for (int i = 1;i < 20;i++) {
        f[u][i] = f[f[u][i - 1]][i - 1];
    }
    for (int v : g[u])if (v != fa)dfs(v, u);
}
int lca(int u, int v) {
    if (dep[u] < dep[v])swap(u, v);
    for (int i = 19;i >= 0;i--) {
        if (dep[f[u][i]] >= dep[v])u = f[u][i];
    }
    if (u == v)return v;
    for (int i = 19;i >= 0;i--) {
        if (f[u][i] != f[v][i])u = f[u][i], v = f[v][i];
    }
    return f[u][0];
}
int dis(int u, int v) {return dep[u] + dep[v] - 2 * dep[lca(u, v)];}

树哈希:
mt19937_64 rng {chrono::steady_clock::now().time_since_epoch().count()};
const ull mask = rng();
ull f(ull x) {
    x ^= mask;
    x ^= x << 13;
    x ^= x >> 7;
    x ^= x << 17;
    x ^= mask;
    return x;
}
vector<int>g[51];
ull getsub(int u,int fa) {
    ull res = 1;
    for (auto v : g[u]) if (v != fa){
        res += f(getsub(v, u));
    }
    return res;
}

ST表LCA:
int dfn[N], lg[N << 1], cnt, id[N << 1], dep[N], st[N << 1][20];

vector<int>g[N];
void dfs(int u, int fa) {
    dfn[u] = ++cnt, id[cnt] = u, dep[u] = dep[fa] + 1;
    for (auto v : g[u]) {
        if (v == fa)continue;
        dfs(v, u);
        id[++cnt] = u;
    }
}
void init() {//预处理
    dfs(1, 0);
    lg[0] = -1;
    for (int i = 1;i <= cnt;i++)st[i][0] = id[i], lg[i] = lg[i >> 1] + 1;
    for (int j = 1;j <= 19;j++) {
        for (int i = 1;cnt - i + 1 >= 1 << j;i++) {
            int f1 = st[i][j - 1], f2 = st[i + (1 << j - 1)][j - 1];
            st[i][j] = dep[f1] < dep[f2] ? f1 : f2;
        }
    }
}
int lca(int u, int v) {//求lca
    u = dfn[u], v = dfn[v];
    if (u > v)u ^= v ^= u ^= v;
    int k = lg[v - u + 1], f1 = st[u][k], f2 = st[v - (1 << k) + 1][k];
    return dep[f1] < dep[f2] ? f1 : f2;
}
