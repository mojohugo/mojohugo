lucas:
C(b, a) % p = C(b / p, a / p) * C(b % p, a % p)%p
ll lucas(ll n, ll m, ll p) {
    if (!m)return 1;
    return C(n % p, m % p, p) * lucas(n / p, m / p, p) % p;
}

void init(ll p) {
    fac[0] = inv[0] = fac[1] = inv[1] = 1;
    for (int i = 2; i < p; i++) {
        fac[i] = fac[i - 1] * i % p;
        inv[i] = (p - p / i) * inv[p % i] % p;
    }
    for (int i = 2;i < p;i++) {
        inv[i] = inv[i] * inv[i - 1] % p;
    }
}

ll fac[N + 5], inv[N + 5];
ll C(ll n, ll m) {
	if (m > n)return 0;
	return fac[n] * inv[n - m] % M * inv[m] % M;
}
ll ksm(ll a, ll b = M - 2) {
	ll s = 1;
    for (; b > 0; b >>= 1, (a *= a) %= M)if (b & 1)(s *= a) %= M;
	return s;

    stl[0][0] = 1;

        for (int j = 1;j <= i;j++) {
            stl[i][j] = (stl[i - 1][j - 1] + j * stl[i - 1][j]) % M;
