## 模板代码

```cpp
int d[N], ans;
char b[N];
void solve() {
    string a;
    cin >> a;
    b[0] = '$', b[1] = '#',d[1] = 1;;
    int k = 1;
    for (int i = 0;i < a.size();i++) {
        b[++k] = a[i], b[++k] = '#';
    } 
    for (int i = 2, l, r = 1;i <= k;i++) {
        if (i <= r)d[i] = min(d[l - i + r], r - i + 1);
        while (b[i - d[i]] == b[i + d[i]])d[i]++;
        if (i + d[i] - 1 > r)l = i - d[i] + 1, r = i + d[i] - 1;
        ans = max(ans, d[i]);
    }
    cout << ans-1;
}

```
