## 模板代码

```cpp
queue<int>q;
for (int i = 1;i <= n;i++) {
    if (!in[i])q.push(i);
}
while (q.size()) {
    int u = q.front();q.pop();
    tp.push_back(u);
    for (auto v : g[u]) {
        if (!--in[v])q.push(v);//无向边就是--in[v]==1
    }
}
```
