## 模板代码

```cpp
 
vector<PII>st[300005];

for (int i = 1;i * i <= 3e5;i++) {
    for (int j = 1;j < i && i * i + j * j <= 3e5;j++) {
        if ((i + j + 1) & 1)continue;
        if (gcd(i, j) != 1)continue;
        int a = i * i - j * j;
        int b = 2 * i * j;
        int c = i * i + j * j;
        if (a > b)swap(a, b);
        for (int x = 1;x * c <= 3e5;x++) {
            st[x * c].push_back({ x * a,x * b });
        }
    }
}

圆上整点:
ll r, ans;
void cal(int d) {
    for (ll s = 1;s * s < r / d;s++) {
        ll t = sqrtl(2 * r / d - s * s);
        if (t * t == 2 * r / d - s * s && gcd(s, t) == 1) {
            // ll x = d * s * t, y = d * (t * t - s * s) / 2;
            ans += 4;
        }
    }
}

void solve() {
    cin >> r;
    for (ll i = 1;i * i <= 2 * r;i++) {
        if ((2 * r) % i == 0) {
            cal(i);
            if (i * i != 2 * r)cal(2 * r / i);
        }
    }
    cout << ans + 4;
}
```
