## 模板代码

```cpp
vector<int> minp, primes;
void sieve(int n) {
    minp.assign(n + 1, 0);
    primes.clear();
    for (int i = 2; i <= n; i++) {
        if (minp[i] == 0) {
            minp[i] = i;
            primes.push_back(i);
        }
        for (auto p : primes) {
            if (i * p > n) {
                break;
            }
            minp[i * p] = p;
            if (p == minp[i]) {
                break;
            }
        }
    }
}

bool notp[N];
int p[N], cnt;
void sieve(int n) {
    for (int i = 2;i <= n;i++) {
        if (!notp[i])p[++cnt] = i;
        for (int j = 1;j <= cnt && i * p[j] <= n;j++) {
            notp[i * p[j]] = 1;
            if (i % p[j] == 0)break;
        }
    }
}
```
