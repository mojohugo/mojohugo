## 模板代码

```cpp
ll tmp[N], a[N];
ll merge(int l, int r) {
    if (l >= r)return 0;
    ll res = 0;
    int mid = l + r >> 1;
    res += merge(l, mid) + merge(mid + 1, r);
    int i = l, j = mid + 1, k = l;
    while (i <= mid && j <= r) {
        if (a[i] <= a[j])tmp[k++] = a[i++];
        else tmp[k++] = a[j++], res += mid - i + 1;
    }
    while (i <= mid)tmp[k++] = a[i++];
    while (j <= r)tmp[k++] = a[j++];
    for (int i = l; i <= r; i++)a[i] = tmp[i];
    return res;
}

```
