struct HLD {
    int n, tot;
    vector<vector<int>> g;
    vector<int> siz, dep, in, out;
    vector<int> top, son, fa;
    HLD(int n) {
        this->n = n;
        this->tot = 0;
        g.resize(n + 1);
        siz.resize(n + 1);
        dep.resize(n + 1);
        top.resize(n + 1);
        son.resize(n + 1);
        fa.resize(n + 1);
        in.resize(n + 1);
        out.resize(n + 1);
    }
    void add(int x, int y) {
        g[x].push_back(y);
        g[y].push_back(x);
    }
    void dfs1(int x) {
        siz[x] = 1;
        dep[x] = dep[fa[x]] + 1;
        for (auto y : g[x]) {
            if (y == fa[x]) continue;
            fa[y] = x;
            dfs1(y);
            siz[x] += siz[y];
            if (siz[y] > siz[son[x]]) {
                son[x] = y;
            }
        }
    }
    void dfs2(int x, int up) {
        top[x] = up, in[x] = ++tot;
        if (son[x]) dfs2(son[x], up);
        for (auto y : g[x]) {
            if (y == fa[x] || y == son[x]) continue;
            dfs2(y, y);
        }
        out[x] = tot;
    }
    int lca(int x, int y) {
        while (top[x] != top[y]) {
            if (dep[top[x]] > dep[top[y]]) {
                x = fa[top[x]];
            }
            else {
                y = fa[top[y]];
            }
        }
        return dep[x] < dep[y] ? x : y;
    }
    int dist(int x, int y) { // 查询两点间距离
        return dep[x] + dep[y] - 2 * dep[lca(x, y)];
    }
    void work(int root = 1) { // 在此初始化
        dfs1(root);
        dfs2(root, root);
    }
};

int p, a[N];
struct HLD {
    int n, tot;
    vector<vector<int>> g;
    vector<int> siz, dep, in, out;
    vector<int> top, son, fa;
    vector<int>w;
    struct node {
        ll sum, tag;
    };
    vector<node>tr;
    HLD(int n) {
        this->n = n;
        this->tot = 0;
        g.resize(n + 1);
        siz.resize(n + 1);
        dep.resize(n + 1);
        top.resize(n + 1);
        son.resize(n + 1);
        fa.resize(n + 1);
        in.resize(n + 1);
        out.resize(n + 1);
        w.resize(n + 1);
        tr.resize(4 * n + 1);
    }
    void add(int x, int y) {
        g[x].push_back(y);
        g[y].push_back(x);
    }
    void dfs1(int x) {
        siz[x] = 1;
        dep[x] = dep[fa[x]] + 1;
        for (auto y : g[x]) {
            if (y == fa[x]) continue;
            fa[y] = x;
            dfs1(y);
            siz[x] += siz[y];
            if (siz[y] > siz[son[x]]) {
                son[x] = y;
            }
        }
    }
    void dfs2(int x, int up) {
        top[x] = up, in[x] = ++tot;
        w[tot] = a[x];
        if (son[x]) dfs2(son[x], up);
        for (auto y : g[x]) {
            if (y == fa[x] || y == son[x]) continue;
            dfs2(y, y);
        }
        out[x] = tot;
    }
    int lca(int x, int y) {
        while (top[x] != top[y]) {
            if (dep[top[x]] > dep[top[y]]) {
                x = fa[top[x]];
            }
            else {
                y = fa[top[y]];
            }
        }
        return dep[x] < dep[y] ? x : y;
    }
    int dist(int x, int y) { // 查询两点间距离
        return dep[x] + dep[y] - 2 * dep[lca(x, y)];
    }
    void work(int root = 1) { // 在此初始化
        dfs1(root);
        dfs2(root, root);
        build(1, 1, n);
    }
    node add(node l, node r) {
        node res;
        res.tag = 0;
        res.sum = l.sum + r.sum;
        return res;
    }
    void pushup(int p) {
        tr[p] = add(tr[lc], tr[rc]);
    }
    void pushdown(int p, int l, int m, int r) {
        if (tr[p].tag) {
            tr[lc].sum += tr[p].tag * (m - l + 1);
            tr[rc].sum += tr[p].tag * (r - m);
            tr[lc].tag += tr[p].tag;
            tr[rc].tag += tr[p].tag;
            tr[p].tag = 0;
        }
    }
    void build(int p, int l, int r) {
        tr[p] = { w[l],0 };
        if (l == r)return;
        int m = l + r >> 1;
        build(lc, l, m);
        build(rc, m + 1, r);
        pushup(p);
    }
    void update(int p, int l, int r, int a, int b, ll k) {
        if (a <= l && r <= b) {
            tr[p].sum += (r - l + 1) * k;
            tr[p].tag += k;
            return;
        }
        int m = l + r >> 1;
        pushdown(p, l, m, r);
        if (a <= m)update(lc, l, m, a, b, k);
        if (b > m)update(rc, m + 1, r, a, b, k);
        pushup(p);
    }
    node qry(int p, int l, int r, int a, int b) {
        if (a <= l && r <= b) {
            return tr[p];
        }
        int m = l + r >> 1;
        pushdown(p, l, m, r);
        if (b <= m)return qry(lc, l, m, a, b);
        if (a > m)return qry(rc, m + 1, r, a, b);
        return add(qry(lc, l, m, a, b), qry(rc, m + 1, r, a, b));
    }
    void addsum(int x, int y, int z) {
        int fx = top[x], fy = top[y];
        while (fx != fy) {
            if (dep[fx] >= dep[fy]) {
                update(1, 1, n, in[fx], in[x], z);
                x = fa[fx], fx = top[x];
            }
            else {
                update(1, 1, n, in[fy], in[y], z);
                y = fa[fy], fy = top[y];
            }
        }
        if (in[x] >= in[y]) {
            update(1, 1, n, in[y], in[x], z);
        }
        else {
            update(1, 1, n, in[x], in[y], z);
        }
    }
    ll qrysum(int x, int y) {
        ll ans = 0;
        int fx = top[x], fy = top[y];
        while (fx != fy) {
            if (dep[fx] >= dep[fy]) {
                ans += qry(1, 1, n, in[fx], in[x]).sum;
                x = fa[fx], fx = top[x];
            }
            else {
                ans += qry(1, 1, n, in[fy], in[y]).sum;
                y = fa[fy], fy = top[y];
            }
        }
        if (in[x] >= in[y]) {
            ans += qry(1, 1, n, in[y], in[x]).sum;
        }
        else {
            ans += qry(1, 1, n, in[x], in[y]).sum;
        }
        return ans;
    }
    void addtr(int x, int z) {
        update(1, 1, n, in[x], out[x], z);
    }
    ll qrytr(int x) {
        return qry(1, 1, n, in[x], out[x]).sum;
    }
};
