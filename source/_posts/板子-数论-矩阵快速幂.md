```cpp
int len;
struct mx {ll a[5][5];};
mx mul(mx &a, mx &b) {
    mx c;
    for (int i = 1; i <= len; i++) {
        for (int j = 1; j <= len; j++) {
            c.a[i][j] = 0;
            for (int k = 1; k <= len; k++) {
                c.a[i][j] = (c.a[i][j] + a.a[i][k] * b.a[k][j] % M + M) % M;
            }
        }
    }
    return c;
}
mx ksm(mx a, ll k) {
    mx res;
    for (int i = 1; i <= len; i++)for (int j = 1; j <= len; j++) {
        res.a[i][j] = (i == j);
    }
    while (k) {
        if (k & 1)res = mul(res, a);
        a = mul(a, a);
        k >>= 1;
    }
    return res;
}
```
