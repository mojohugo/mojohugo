## 模板代码

```cpp
template <typename T>
int Min_show(T& A) {
    int i = 0, j = 1, k = 0;
    int n = A.size();
    while (i < n && j < n && k < n) {
        if (A[(i + k) % n] == A[(j + k) % n])k++;
        else {
            if (A[(i + k) % n] > A[(j + k) % n]) i += k + 1;
            else j += k + 1;
            if (i == j)i++;
            k = 0;
        }
    }
    return min(i, j);
}
```
