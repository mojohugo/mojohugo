## ACAM

> 来源：`板子/字符串/ACAM.txt`

```cpp
int dep[N], ch[N][26], cnt[N], idx, fail[N];
vector<int>v[N], g[N];
void insert(string& s, int id) {
    int p = 0;
    for (auto c : s) {
        int j = c - 'a';
        if (!ch[p][j]) {
            ch[p][j] = ++idx;
            dep[idx] = dep[p] + 1;
        }
        p = ch[p][j];
        v[p].push_back(id);
    }
    cnt[p]++;
}
void build() {
    queue<int>q;
    for (int i = 0;i < 26;i++) {
        if (ch[0][i])q.push(ch[0][i]);
    }
    while (q.size()) {
        int u = q.front();q.pop();
        for (int i = 0;i < 26;i++) {
            int v = ch[u][i];
            if (v) {
                fail[v] = ch[fail[u]][i], q.push(v);
            }
            else ch[u][i] = ch[fail[u]][i];
        }
    }
    for (int i = 1;i <= idx;i++)g[fail[i]].push_back(i);
}
void dfs(int u) {
    for (auto v : g[u]) {
        dfs(v);
    }
}

```

```cpp
void KMP(string s, string t) {
    int n = s.size(), m = t.size();
    vector<int> nxt(m + 1);
    s = '-' + s;
    t = '-' + t;
    for (int i = 2, j = 0; i <= m; i++) {
        while (j && t[i] != t[j + 1]) j = nxt[j];
        if (t[i] == t[j + 1]) j++;
        nxt[i] = j;
    }
    for (int i = 1, j = 0; i <= n; i++) {
        while (j && s[i] != t[j + 1]) j = nxt[j];
        if (s[i] == t[j + 1]) j++;
        if (j == m) {
            cout << i - m + 1;
            j = nxt[j];
        }
    }
}
```

## 马拉车

> 来源：`板子/字符串/马拉车.txt`

```cpp
int d[N], ans;
char b[N];
void solve() {
    string a;
    cin >> a;
    b[0] = '$', b[1] = '#',d[1] = 1;;
    int k = 1;
    for (int i = 0;i < a.size();i++) {
        b[++k] = a[i], b[++k] = '#';
    } 
    for (int i = 2, l, r = 1;i <= k;i++) {
        if (i <= r)d[i] = min(d[l - i + r], r - i + 1);
        while (b[i - d[i]] == b[i + d[i]])d[i]++;
        if (i + d[i] - 1 > r)l = i - d[i] + 1, r = i + d[i] - 1;
        ans = max(ans, d[i]);
    }
    cout << ans-1;
}
```

```cpp
int ch[N * 20][2], idx;

void insert(int x) {
    int p = 0;
    for (int i = 19;i >= 0;i--) {
        int t = x >> i & 1;
        if (!ch[p][t])ch[p][t] = ++idx;
        p = ch[p][t];
    }
}

int qry(int x) {
    int p = 0;
    int res = 0;
    for (int i = 19;i >= 0;i--) {
        int t = x >> i & 1;
        if (ch[p][!t]) {
            res = res * 2 + 1;
            p = ch[p][!t];
        }
        else {
            p = ch[p][t];
            res *= 2;
        }
    }
    return res;
}
```

## 字符串哈希

> 来源：`板子/字符串/字符串哈希.txt`

```cpp
const int m[2] = { 1000000007,1000000009 };
const int base[2] = { 131,13331 };
 
struct hash {
    int h1[N][2], p[N][2], h2[N][2];   
    void init(string &s) {
        p[0][0] = p[0][1] = 1;
        int n = s.size();
        for (int i = 0;i < n;i++) {
            for (int j = 0;j < 2;j++) {
                h1[i + 1][j] = (1LL*h1[i][j] * base[j] + s[i]) % m[j];
                h2[n - 1 - i][j] = (1LL*h2[n - i][j] * base[j] + s[n - i - 1])%m[j];
                p[i + 1][j] = 1LL*p[i][j] * base[j] % m[j];
            }         
        }
    }
    PII get1(int l, int r) {
        int a = (h1[r][0] - 1LL*h1[l - 1][0] * p[r - l + 1][0] % m[0] + m[0]) % m[0];
        int b = (h1[r][1] - 1LL*h1[l - 1][1] * p[r - l + 1][1] % m[1] + m[1]) % m[1];
        return { a,b };
    }
    PII get2(int l, int r) {
        int a = (h2[l - 1][0] - 1LL*h2[r][0] * p[r - l + 1][0] % m[0] + m[0]) % m[0];
        int b = (h2[l - 1][1] - 1LL*h2[r][1] * p[r - l + 1][1] % m[1] + m[1]) % m[1];
        return { a,b };
    }
};

struct Hash2D {
    int n, m;
    ull BX, BY;
    vector<ull> px, py;
    vector<vector<ull>> H;

    Hash2D(const vector<string>& a, ull bx = 131, ull by = 133) {
        n = a.size();
        m = n ? (int)a[0].size() : 0;
        BX = bx; BY = by;

        px.assign(n + 1, 1);
        py.assign(m + 1, 1);
        for (int i = 1; i <= n; i++) px[i] = px[i - 1] * BX;
        for (int j = 1; j <= m; j++) py[j] = py[j - 1] * BY;

        H.assign(n + 1, vector<ull>(m + 1, 0));

        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                ull val = (ull)(a[i - 1][j - 1] + 1);
                H[i][j] = H[i - 1][j] * BX + H[i][j - 1] * BY - H[i - 1][j - 1] * BX * BY + val;
            }
        }
    }

    // 查询子矩阵 [x1..x2][y1..y2]
    ull get(int x1, int y1, int x2, int y2) {
        int dx = x2 - x1 + 1;
        int dy = y2 - y1 + 1;
        ull res = H[x2][y2];
        res -= H[x1 - 1][y2] * px[dx];
        res -= H[x2][y1 - 1] * py[dy];
        res += H[x1 - 1][y1 - 1] * px[dx] * py[dy];
        return res;
    }
};

struct Hash2D {
    int n, m;
    int BX, BY;
    static const int MOD1 = 1000000007;
    static const int MOD2 = 1000000009;

    vector<int> px1, px2, py1, py2;
    vector<vector<int>> H1, H2;

    Hash2D(const vector<string>& a, int bx = 131, int by = 133) {
        n = a.size();
        m = n ? a[0].size() : 0;
        BX = bx; BY = by;

        px1.assign(n + 1, 1);
        px2.assign(n + 1, 1);
        py1.assign(m + 1, 1);
        py2.assign(m + 1, 1);

        for (int i = 1; i <= n; i++) {
            px1[i] = 1LL * px1[i - 1] * BX % MOD1;
            px2[i] = 1LL * px2[i - 1] * BX % MOD2;
        }
        for (int j = 1; j <= m; j++) {
            py1[j] = 1LL * py1[j - 1] * BY % MOD1;
            py2[j] = 1LL * py2[j - 1] * BY % MOD2;
        }

        H1.assign(n + 1, vector<int>(m + 1, 0));
        H2.assign(n + 1, vector<int>(m + 1, 0));

        // 构建二维前缀哈希 (1-indexed)
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                int val = a[i - 1][j - 1] + 1;

                H1[i][j] = ((1LL * H1[i - 1][j] * BX % MOD1
                    + 1LL * H1[i][j - 1] * BY % MOD1
                    - 1LL * H1[i - 1][j - 1] * BX % MOD1 * BY % MOD1
                    + val) % MOD1 + MOD1) % MOD1;

                H2[i][j] = ((1LL * H2[i - 1][j] * BX % MOD2
                    + 1LL * H2[i][j - 1] * BY % MOD2
                    - 1LL * H2[i - 1][j - 1] * BX % MOD2 * BY % MOD2
                    + val) % MOD2 + MOD2) % MOD2;
            }
        }
    }

    // 获取子矩阵 [x1..x2][y1..y2] 的哈希
    pair<int, int> get(int x1, int y1, int x2, int y2) {
        int dx = x2 - x1 + 1;
        int dy = y2 - y1 + 1;

        int res1 = H1[x2][y2];
        res1 = (res1 - 1LL * H1[x1 - 1][y2] * px1[dx] % MOD1 + MOD1) % MOD1;
        res1 = (res1 - 1LL * H1[x2][y1 - 1] * py1[dy] % MOD1 + MOD1) % MOD1;
        res1 = (res1 + 1LL * H1[x1 - 1][y1 - 1] * px1[dx] % MOD1 * py1[dy] % MOD1) % MOD1;

        int res2 = H2[x2][y2];
        res2 = (res2 - 1LL * H2[x1 - 1][y2] * px2[dx] % MOD2 + MOD2) % MOD2;
        res2 = (res2 - 1LL * H2[x2][y1 - 1] * py2[dy] % MOD2 + MOD2) % MOD2;
        res2 = (res2 + 1LL * H2[x1 - 1][y1 - 1] * px2[dx] % MOD2 * py2[dy] % MOD2) % MOD2;

        return { res1, res2 };
    }
};
```

## 最小表示法

> 来源：`板子/字符串/最小表示法.txt`

```cpp
template <typename T>
int Min_show(T& A) {
    int i = 0, j = 1, k = 0;
    int n = A.size();
    while (i < n && j < n && k < n) {
        if (A[(i + k) % n] == A[(j + k) % n])k++;
        else {
            if (A[(i + k) % n] > A[(j + k) % n]) i += k + 1;
            else j += k + 1;
            if (i == j)i++;
            k = 0;
        }
    }
    return min(i, j);
}
```
