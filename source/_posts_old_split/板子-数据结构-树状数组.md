## 模板代码

```cpp
template <typename T>
struct FWT{
    int n;
    vector<T> a;   
    FWT(int n_ = 0) {
        init(n_);
    }
    void init(int n_) {
        n = n_;
        a.assign(n + 1, T{});
    }
    void add(int x, const T &v) {
        for (int i = x; i <= n; i += i & -i) {
            a[i] += v;
        }
    }
    T qry(int x) {
        T ans{};
        for (int i = x; i > 0; i -= i & -i) {
            ans += a[i];
        }
        return ans;
    }
    T qry(int l, int r) {
        return qry(r) - qry(l - 1);
    }
    int kth(const T &k) {
        int x = 0;
        T cur{};
        for (int i = 1 << __lg(n); i > 0; i /= 2) {
            if (x + i <= n && cur + a[x + i] < k) {
                x += i;
                cur += a[x];
            }
        }
        return x + 1;
    }
    int mex() {//注意不能add(0)，算出来mex-=1
        int x = 0;
        T cur{};
        for (int i = 1 << __lg(n); i > 0; i /= 2) {
            if (x + i <= n && cur + a[x + i] == x + i) {
                x += i;
                cur += a[x];
            }
        }
        return x + 1;
    }
};

```
