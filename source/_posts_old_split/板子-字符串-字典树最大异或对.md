## 模板代码

```cpp
int ch[N * 20][2], idx;

void insert(int x) {
    int p = 0;
    for (int i = 19;i >= 0;i--) {
        int t = x >> i & 1;
        if (!ch[p][t])ch[p][t] = ++idx;
        p = ch[p][t];
    }
}

int qry(int x) {
    int p = 0;
    int res = 0;
    for (int i = 19;i >= 0;i--) {
        int t = x >> i & 1;
        if (ch[p][!t]) {
            res = res * 2 + 1;
            p = ch[p][!t];
        }
        else {
            p = ch[p][t];
            res *= 2;
        }
    }
    return res;
}
```
