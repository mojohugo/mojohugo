## 模板代码

```cpp

区间加,区间询问:
#define lc p<<1
#define rc p<<1|1
ll a[N], n, m;
struct node {
    ll sum, tag;
}tr[N * 4];

node add(node l, node r) {
    node res;
    res.tag = 0;
    res.sum = l.sum + r.sum;
    return res;
}

void pushup(int p) {
    tr[p] = add(tr[lc], tr[rc]);
}

void pushdown(int p, int l, int m, int r) {
    if (tr[p].tag) {
        tr[lc].sum += tr[p].tag * (m - l + 1);
        tr[rc].sum += tr[p].tag * (r - m);
        tr[lc].tag += tr[p].tag;
        tr[rc].tag += tr[p].tag;
        tr[p].tag = 0;
    }
}

void build(int p, int l, int r) {
    tr[p] = { a[l],0 };
    if (l == r)return;
    int m = l + r >> 1;
    build(lc, l, m);
    build(rc, m + 1, r);
    pushup(p);
}

void update(int p, int l, int r,int a,int b, ll k) {
    if (a <= l && r <= b) {
        tr[p].sum += (r - l + 1) * k;
        tr[p].tag += k;
        return;
    }
    int m = l + r >> 1;
    pushdown(p, l, m, r);
    if (a <= m)update(lc, l, m, a, b, k);
    if (b > m)update(rc, m + 1, r, a, b, k);
    pushup(p);
}

node qry(int p, int l, int r, int a, int b) {
    if (a <= l && r <= b) {
        return tr[p];
    }
    int m = l + r >> 1;
    pushdown(p, l, m, r);
    if (b <= m)return qry(lc, l, m, a, b);
    if (a > m)return qry(rc, m + 1, r, a, b);
    return add(qry(lc, l, m, a, b), qry(rc, m + 1, r, a, b));
}
```
