## 模板代码

```cpp
区间第k小:
#define lc(x) tr[x].ch[0]
#define rc(x) tr[x].ch[1]
struct node {
    int ch[2], s;
}tr[N * 24];
int root[N], idx;

int n, m;
void build(int& x, int l, int r) {
    x = ++idx;
    if (l == r)return;
    int m = l + r >> 1;
    build(lc(x), l, m);
    build(rc(x), m + 1, r);
}

void insert(int x, int& y, int l, int r,int v) {
    y = ++idx;tr[y] = tr[x]; tr[y].s++;
    if (l == r)return;
    int m = l + r >> 1;
    if (v <= m)insert(lc(x), lc(y), l, m, v);
    else insert(rc(x), rc(y), m + 1, r, v);
}

int qry(int x, int y, int l, int r, int k) {
    if (l == r)return l;
    int m = l + r >> 1;
    int s = tr[lc(y)].s - tr[lc(x)].s;
    if (k <= s)return qry(lc(x), lc(y), l, m, k);
    else return qry(rc(x), rc(y), m + 1, r, k - s);
}
int qry(int l, int r, int k) {
    return qry(root[l - 1], root[r], 1, n, k);
}

void insert(int i, int v) {
    insert(root[i - 1], root[i], 1, n, v);
}

int a[N];

build(root[0], 1, n);
vector<int>v(n);
for (int i = 1;i <= n;i++) {
    cin >> a[i];
    v[i - 1] = a[i];
}
sort(v.begin(), v.end());
v.erase(unique(v.begin(), v.end()), v.end());
for (int i = 1;i <= n;i++) {
    int it = lower_bound(v.begin(), v.end(), a[i]) - v.begin() + 1;
    insert(i, it);
}
while (m--) {
    int l, r, k;
    cin >> l >> r >> k;
    cout << v[qry(l, r, k)-1] << endl;
}
```
